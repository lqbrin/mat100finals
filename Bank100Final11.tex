\newproblem{23a}
{
	Let $C(w)=\dfrac{1}{5}w-3$. Find $C(7)$ and $C(a+1).$  Be sure to simplify your answers.
	
}
{
	\begin{tabular}{l r}
	$C(7)=-1.6$ & 2 pts\\
	$C(a+1)=\dfrac{1}{5}(a+1)-3$ & 1 additional point\\
	$C(a+1)=\dfrac{1}{5}a-\dfrac{14}{5}$ & 1 additional point
	\end{tabular}
}
