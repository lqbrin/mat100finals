{

}

	\begin{tabular}{l r}
	$\frac{\sqrt{3}+3}{\sqrt{3}-3} \times \frac{\sqrt{3}+3}{\sqrt{3}+3}$ & 2 pts  to here\\
	$\frac{3+3\sqrt{3}+3\sqrt{3}+9}{3-9}$ & 3 pts to here\\
	$\frac{12+6\sqrt{3}}{-6}$ \\
	$-2-\sqrt{3}$ & 4 pts to here \\
	\end{tabular}
}
