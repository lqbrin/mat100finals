\newproblem{15.5a}
{
	Solve by using the square root property. $\displaystyle (2x-10)^2=7$.
}
{
	\begin{tabular}{l r}
	$2x-10=\pm\sqrt{7}$ & 1 pt to here\\
	$2x=10+\sqrt{7}$ &  \\
	$x=\dfrac{10+\sqrt{7}}{2}$ & 2 pts to here \\
	$2x=10-\sqrt{7}$ & 3 pts to here\\
	$x=\dfrac{10-\sqrt{7}}{2}$ & 4 pts \\
	\end{tabular}
}
