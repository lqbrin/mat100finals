\newproblem{07a}
{
	Simplify $\displaystyle \dfrac{\dfrac{a}{3b}-\dfrac{1}{2}}{\dfrac{7}{3b}-\dfrac{4}{a}}$.
}
{
	\begin{tabular}{l r}
	Method 1 \\
	$\dfrac{6ab}{6ab}\left(\dfrac{\dfrac{a}{3b}-\dfrac{1}{2}}{\dfrac{7}{3b}-\dfrac{4}{a}}\right)$& 2 pts to here\\
	$\dfrac{2a^2-3ab}{14a-24b}$ & 4 pts to here\\ \\
	Method 2\\
	$\dfrac{\dfrac{2a}{6b}-\dfrac{3b}{6b}}{\dfrac{7a}{3ab}-\dfrac{12b}{3ab}}$ & 1 pt to here\\
	$\dfrac{\dfrac{2a-3b}{6b}}{\dfrac{7a-12b}{3ab}}$ & 2 pts to here\\
	$\dfrac{2a-3b}{6b} \times \dfrac{3ab}{7a-12b}$ & 3 pts to here\\
	$\dfrac{a(2a-3b)}{2(7a-12b)}$ or $\dfrac{2a^2-3ab}{14a-24b}$ & 4 pts to here
	\end{tabular}
}
