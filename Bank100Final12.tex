\newproblem{12a}
{
	Solve $\displaystyle \sqrt{x+8}-8=x$.
}
{
	\begin{tabular}{l r}
	$\sqrt{x+8}=x+8$ & \\
	$x+8=x^2+16x+64$ & 1  pt to here\\
	$0=x^2+15x+56$ & 2 pts to here\\
	$0=(x+7)(x+8)$ & 3 pts to here \\
	$x=-7$ and $x=-8$ & 4 pts to here
	\end{tabular}
}
