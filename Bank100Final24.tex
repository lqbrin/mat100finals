\newproblem{24a}
{
	Find the difference quotient of $f$; that is find $\displaystyle\frac{f(x+h)-f(x)}{h}$. Assume $h\neq 0$. $f(x)=5x+8$.
}
{
	\begin{tabular} {l r}
	$f(x+h)=5x+5h+8$ & 1 pt \\
	$\frac{(5x+5h+8)-(5x+8)}{h}$ & 2 pts to here\\
	$\frac{5h}{h}$ & 3 pts to here\\
	$5$ & 4 pts to here
	\end{tabular}
}
