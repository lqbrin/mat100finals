\newproblem{08.5a}
{
	Solve $\displaystyle \frac{5}{2}=\frac{2x+3}{x+5}+4$.
}
{
	\begin{tabular}{l r}
	$5(x-2)=2(2x+3)+4(2)(x+5)$ & 1 pt to here \\
	$5x+25=4x+6+8x+40$ & 2 pts to here \\
	$5x+25=12x+46$ & 3 pts to here\\
	$-3=x$ & 4 pts to here
	\end{tabular}
}
