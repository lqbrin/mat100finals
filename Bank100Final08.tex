\newproblem{08a}
{
	Solve $\displaystyle \frac{2}{x^2-4}+\frac{5}{x+2}=\frac{2}{x-2}$.
}
{
	\begin{tabular}{l r}
	$2 + 5(x-2)=2(x+2)$ & 2 pts to here \\
	$5x-8=2x+4$ & 3 pts to here\\
	$x=4$ & 4 pts to here
	\end{tabular}
}
