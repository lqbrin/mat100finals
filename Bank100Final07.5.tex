\newproblem{07.5a}
{
	Simplify $ \dfrac{\dfrac{3}{x}}{\dfrac{5}{x}+\dfrac{3}{x^2}}$.
}
{
	\begin{tabular}{l r}
	Method 1 \\
	$\dfrac{x^2}{x^2}\left(\dfrac{\dfrac{3}{x}}{\dfrac{5}{x}+\dfrac{3}{x^2}}\right)$& 2 pts to here\\
	$\dfrac{3x}{5x+3}$ & 4 pts to here\\
	Method 2\\
	$\dfrac{\dfrac{3}{x}}{\dfrac{5x}{x^2}+\dfrac{3}{x^2}}$ & 1 pt to here\\
	$\dfrac{\dfrac{3}{x}}{\dfrac{5x+3}{x^2}}$ & 2 pts to here\\
	$\dfrac{3}{x} \times \dfrac{x^2}{5x+3}$ & 3 pts to here\\
	$\dfrac{3x}{5x+3}$ & 4 pts to here
	\end{tabular}
}
