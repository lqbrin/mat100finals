\newproblem{04.5a}
{
	Simplify $\displaystyle \frac{15x^3}{5x-10}\div\frac{10x}{(x-2)(x+2)}$.
}
{
	\begin{tabular}{l r}
	$\frac{15x^3}{5(x-2)}\div\frac{10x}{(x-2)(x+2)}$ & 1 pt \\
	$\frac{15x^3}{5(x-2)} \times \frac{(x-2)(x+2)}{10x}$ 3 pts to here\\
	$\frac{3x^2(x+2)}{10}$ & 5 pts to here
	\end{tabular}
}
