\newproblem{23.5a}
{
	Let $h(t)=3t^2-9t+2$. Find $h(3)$ and $h(-2).$  Be sure to simplify your answers.
	
}
{
	\begin{tabular}{l r}
	$h(3)=2$ & 2 pts\\
	$h(-2)=3(-2)^2-9(-2)+2$ & 1 additional point\\
	$h(-2)=32$ & 1 additional point
	\end{tabular}
}
